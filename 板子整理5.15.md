``` c++
int qpow (int a, int b)
{
    int res = 1;
    while (b) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
```

### 矩阵快速幂
``` c++
const int mod = 998244353;

struct mtx {
    int n, m;
    vector <vector <int> > a;

    mtx (int n = 0, int m = 0) : n(n), m(m) {
        a.assign(n + 1, vector <int> (m + 1, 0));
    }

    static mtx I (int n) {
        mtx I(n, n);
        for (int i = 1; i <= n; i++) {
            I.a[i][i] = 1;
        }
        return I;
    }

    mtx operator * (const mtx& other) const {
        mtx res(n, other.m);
        for (int i = 1; i <= n; i++) {
            for (int k = 1; k <= m; k++) {
                if (a[i][k] == 0) continue;
                for (int j = 1; j <= other.m; j++) {
                    res.a[i][j] = (res.a[i][j] + a[i][k] * other.a[k][j] % mod) % mod;
                }
            }
        }
        return res;
    }
};

mtx qpow (mtx a, int b) {
    mtx res = mtx::I(a.n);
    while (b) {
        if (b & 1) res = res * a;
        a = a * a;
        b >>= 1;
    }
    return res;
}
```

#### 挡板法

##### 允许空结论
`x1 + x2 + ... + xk = n, xi >= 0` 的解数为：

$$
\binom{n + k - 1}{k - 1}
$$

##### 不允许空结论
`x1 + x2 + ... + xk = n, xi >= 1` 的解数为：

$$
\binom{n - 1}{k - 1}
$$
（前提 `n >= k`，否则为 0）

``` c++
int exgcd (int a, int b, int &x, int &y)
{
    if (b == 0) {
        x = 1;
        y = 0;
        return a;
    }
    int g = exgcd(b, a % b, y, x);
    y -= a / b * x;
```

``` c++
struct BIT {
    int n;
    vector <int> bit;

    BIT (int n = 0) : n(n) {
        bit.assign(n + 1, 0);
    }

    int lowbit (int x) {
        return x & -x;
    }

    int sum (int idx) {
        int res = 0;
        for (; idx > 0; idx -= lowbit(idx)) res += bit[idx];
        return res; 
    }

    void point_add (int idx, int diff) {
        for (; idx <= n; idx += lowbit(idx)) bit[idx] += diff;
    }

    int point_query (int idx) {
        return sum(idx) - sum(idx - 1);
    }
```

### 并查集
``` c++
struct DSU {
    vector<int> fa, rk, sz;
    int comps;

    DSU(int n = 0) {
        init(n);
    }

    void init(int n) {
        fa.resize(n + 1);
        rk.assign(n + 1, 1);
        sz.assign(n + 1, 1);
        iota(fa.begin(), fa.end(), 0);
        comps = n;
    }

    int find(int x) {
        return x == fa[x] ? x : (fa[x] = find(fa[x]));
    }

    bool merge(int i, int j) {
        int x = find(i), y = find(j);
        if (x == y) return false;
        if (rk[x] < rk[y]) swap(x, y);
        fa[y] = x;
        sz[x] += sz[y];
        if (rk[x] == rk[y]) rk[x]++;
        comps--;
        return true;
    }

    int size(int x) {
        return sz[find(x)];
    }
};
```

### 线段树
``` c++
struct SegTree {
    struct node {
        int l, r;
        int s, lz;
        bool tag;
    };
    
    int n;
    vector <int> a;
    vector <node> tr;

    SegTree (int n = 0) : n(n){
        a.assign(n + 1, 1);
        tr.assign(4 * n + 10, {0, 0, 0, 0});
    }

    void push_up (int p) {
        tr[p].s = tr[p << 1].s + tr[p << 1 | 1].s;
    }

    void apply (int p, int t) {
        tr[p].s = (tr[p].r - tr[p].l + 1) * t;
        tr[p].lz = t;
        tr[p].tag = true;
    }

    void push_down (int p) {
        if (tr[p].tag) {
            apply(p << 1, tr[p].lz);
            apply(p << 1 | 1, tr[p].lz);
            tr[p].tag = false;
        }
    }

    void build (int p, int l, int r) {
        tr[p] = {l, r, 0, 0};
        if (l == r) {
            tr[p].s = a[l];
            return;
        }
        int mid = (l + r) >> 1;
        build(p << 1, l, mid);
        build(p << 1 | 1, mid + 1, r);
        push_up(p);
    }

    void modify (int p, int l, int r, int t) {
        if (l <= tr[p].l && tr[p].r <= r) {
            apply(p, t);
            return;
        }
        push_down(p);
        int mid = (tr[p].l + tr[p].r) >> 1;
        if (l <= mid) modify(p << 1, l, r, t);
        if (mid < r) modify(p << 1 | 1, l, r, t);
        push_up(p);
    }

    int query (int p, int l, int r) {
        if (l <= tr[p].l && tr[p].r <= r) {
            return tr[p].s;
        }
        int res = 0;
        push_down(p);
        int mid = (tr[p].l + tr[p].r) >> 1;
        if (l <= mid) res += query(p << 1, l, r);
        if (mid < r) res += query(p << 1 | 1, l, r);
        return res;
    }
};
```

### 动态开点线段树
``` c++
struct DynamicSegTree {
    struct node {
        int ls, rs;
        int s, lz;
        bool tag;
    };

    vector <node> tr;

    DynamicSegTree () {
        tr.reserve(8e6);
        tr.push_back({0, 0, 0, 0, false});
        tr.push_back({0, 0, 0, 0, false});
    }

    int nw () {
        tr.push_back({0, 0, 0, 0, false});
        return tr.size() - 1;
    }

    void push_up (int p) {
        tr[p].s = 0;
        if (tr[p].ls) tr[p].s = (tr[p].s + tr[tr[p].ls].s) % mod;
        if (tr[p].rs) tr[p].s = (tr[p].s + tr[tr[p].rs].s) % mod;
    }

    void apply (int p, int l, int r, int t) {
        tr[p].s = (tr[p].s + (r - l + 1) % mod * t % mod) % mod;
        tr[p].lz = (tr[p].lz + t) % mod;
        tr[p].tag = 1;
    }

    void push_down (int p, int l, int r) {
        if (!tr[p].tag || l == r) return;
        int mid = l + ((r - l) >> 1);
        if (!tr[p].ls) tr[p].ls = nw();
        if (!tr[p].rs) tr[p].rs = nw();
        apply(tr[p].ls, l, mid, tr[p].lz);
        apply(tr[p].rs, mid + 1, r, tr[p].lz);
        tr[p].tag = 0;
        tr[p].lz = 0;
    }

    void modify (int p, int l, int r, int ql, int qr, int t) {
        if (ql <= l && r <= qr) {
            apply(p, l, r, t);
            return;
        }
        push_down(p, l, r);
        int mid = l + ((r - l) >> 1);
        if (ql <= mid) {
            if (!tr[p].ls) tr[p].ls = nw();
            modify(tr[p].ls, l, mid, ql, qr, t);
        }
        if (mid < qr) {
            if (!tr[p].rs) tr[p].rs = nw();
            modify(tr[p].rs, mid + 1, r, ql, qr, t);
        }
        push_up(p);
    }

    int query (int p, int l, int r, int ql, int qr) {
        if (!p || ql > r || qr < l) return 0;
        if (ql <= l && r <= qr) {
            return tr[p].s;
        }
        push_down(p, l, r);
        int res = 0;
        int mid = l + ((r - l) >> 1);
        if (ql <= mid) res = (res + query(tr[p].ls, l, mid, ql, qr)) % mod;
        if (mid < qr) res = (res + query(tr[p].rs, mid + 1, r, ql, qr)) % mod;
        return res;
    }
};
```

### 可持久化线段树
``` c++
struct PSegTree {
    struct node {
        int ls, rs;
        int s;
    };

    int n;
    vector <int> a;
    vector <node> tr;
    vector <int> rt;

    PSegTree (int n = 0, int m = 0) : n(n) {
        a.assign(n + 1, 0);
        tr.reserve(2 * n + (m + 5) * 25);
        tr.push_back({0, 0, 0});
        rt.assign(m + 1, 0);
    }

    int nw () {
        tr.push_back({0, 0, 0});
        return tr.size() - 1;
    }

    int clone (int p) {
        tr.push_back(tr[p]);
        return tr.size() - 1;
    }

    void push_up (int p) {
        tr[p].s = tr[tr[p].ls].s + tr[tr[p].rs].s;
    }

    int build (int l, int r) {
        int p = nw();
        if (l == r) {
            tr[p].s = a[l];
            return p;
        }
        int mid = l + ((r - l) >> 1);
        tr[p].ls = build(l, mid);
        tr[p].rs = build(mid + 1, r);
        push_up(p);
        return p;
    }
    
    int modify (int p, int l, int r, int x, int t) {
        int q = clone(p);
        if (l == r) {
            tr[q].s = t;
            return q;
        }
        int mid = l + ((r - l) >> 1);
        if (x <= mid) {
            tr[q].ls = modify(tr[p].ls, l, mid, x, t);
        }else {
            tr[q].rs = modify(tr[p].rs, mid + 1, r, x, t);
        }
        push_up(q);
        return q;
    }

    int query (int p, int l, int r, int ql, int qr) {
        if (!p || l > qr || r < ql) return 0;
        if (ql <= l && r <= qr) return tr[p].s;
        int mid = l + ((r - l) >> 1);
        int res = 0;
        if (ql <= mid) {
            res += query(tr[p].ls, l, mid, ql, qr);
        }
        if (mid < qr) {
            res += query(tr[p].rs, mid + 1, r, ql, qr);
        }
        return res;
    }

};
```

### LCA & 树上倍增
``` c++
void solve ()
{
    int n, q;
    cin >> n >> q;
    vector <int> ind(n + 1);
    vector <vector <int> > e(n + 1);
    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        e[u].push_back(v);
        e[v].push_back(u);
        ind[u]++;
        ind[v]++;
    }

    vector <int> fa(n + 1);
    vector <int> dep(n + 1);
    vector <int> pre(n + 1);
    fa[1] = 0;
    dep[0] = -1;

    auto dfs = [&] (auto self, int f, int u) -> void {
        pre[u] = pre[f] + ind[u];
        fa[u] = f;
        dep[u] = dep[f] + 1;
        
        for (auto v : e[u]) {
            if (v == f) continue;
            self(self, u, v);
        }
    };
    dfs(dfs, 0, 1);

    int m = __lg(n);
    vector <vector <int> > st(n + 1, vector <int> (m + 1));
    for (int i = 1; i <= n; i++) {
        st[i][0] = fa[i];
    }
    for (int k = 1; k <= m; k++) {
        for (int i = 1; i <= n; i++) {
            st[i][k] = st[st[i][k - 1]][k - 1];
        }
    }

    auto lca = [&] (int u, int v) -> int {
        if (dep[u] < dep[v]) swap(u, v);
        if (dep[u] > dep[v]) {
            for (int k = m; k >= 0; k--) {
                if (dep[st[u][k]] >= dep[v]) {
                    u = st[u][k];
                }
            }
        }

        if (u == v) return u;
        for (int k = m; k >= 0; k--) {
            if (st[u][k] != st[v][k]) {
                u = st[u][k];
                v = st[v][k];
            }
        }

        return fa[u];
    };

    while (q--) {
        int u, v;
        cin >> u >> v;
        int x = lca(u, v);
        int res = pre[u] + pre[v] - 2 * pre[x] + ind[x];
        cout << res << '\n';
    }
}
```

### 二进制优化多重背包
```cpp
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, W;
    cin >> N >> W;

    vector<int> dp(W + 1, INF); // dp表示长度为W的时候的最小的数量cnt
    dp[0] = 0; // 1. 我们dp求的是最小值 2. 长度W不一定能够到达 -> 初始化dp[0] = 0, 其他为INF

    for (int i = 0; i < N; ++i) {
        int w, cnt;
        cin >> w >> cnt;

        for (int k = 1; cnt > 0; k <<= 1) {
            int take = min(k, cnt);
            cnt -= take;

            // 二进制优化
            // 因为每个被拆分出来的组只能选择一次, 相当于01背包
            int weight = take * w; // 这一组的总重量
            int cost = take;       // 这一组的总代价

            // 01背包, 最小化版(逻辑都一样)
            for (int j = W; j >= weight; --j) {
                if (dp[j - weight] != INF) {
                    dp[j] = min(dp[j], dp[j - weight] + cost);
                }
            }
        }
    }

    cout << (dp[W] == INF ? -1 : dp[W]) << '\n';
    return 0;
}
```

```cpp
    vector <vector <int> > dp (n + 1, vector <int> (m + 1, -INF));
    
    vector <int> sz (n + 1);

    auto dfs = [&] (auto self, int u) -> void {
        if (u > n - m) {
            dp[u][1] = val[u];
            dp[u][0] = 0;
            sz[u] = 1;
            return;
        }

        dp[u][0] = 0;

        for (auto v : g[u]) {
            self (self, v);

            for (int i = sz[u]; i >= 0; i --) {
                for (int j = sz[v]; j >= 1; j --) {
                    dp[u][i + j] = max (dp[u][i + j], dp[u][i] - cs[u][v] + dp[v][j]);
                }
            }

            sz[u] += sz[v];
        }
    };

    dfs (dfs, 1);
}
```

``` c++
struct Point {
    int x, y;

    Point() {}
    Point(int _x, int _y) : x(_x), y(_y) {}

    Point operator + (const Point& other) const {
        return Point(x + other.x, y + other.y);
    }

    Point operator - (const Point& other) const {
        return Point(x - other.x, y - other.y);
    }

    bool operator < (const Point& other) const {
        if (x != other.x) return x < other.x;
        return y < other.y;
    }

    bool operator == (const Point& other) const {
        return x == other.x && y == other.y;
    }
};

// 叉积 a x b
int cross(Point a, Point b) {
    return a.x * b.y - a.y * b.x;
}

// 三角形两倍面积
int triangleArea2(Point a, Point b, Point c) {
    return abs(cross(b - a, c - a));
}

// 多边形两倍面积
int polygonArea2(const vector<Point>& p) {
    int n = p.size();
    int sum = 0;

    for (int i = 0; i < n; i++) {
        int j = (i + 1) % n;
        sum += cross(p[i], p[j]);
    }

    return abs(sum);
}

// 输出面积，支持 .5
void printArea(int area2) {
    cout << area2 / 2;
    if (area2 % 2) cout << ".5";
    cout << '\n';
}
```

``` c++
const int mod = 998244353;

int qpow(int a, int b) {
    int res = 1;
    while (b) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

vector<int> get_inv(const vector<int> &a) {
    int n = a.size();

    vector<int> pre(n + 1, 1);
    vector<int> inv(n);

    for (int i = 0; i < n; i++) {
        pre[i + 1] = pre[i] * a[i] % mod;
    }

    int x = qpow(pre[n], mod - 2);

    for (int i = n - 1; i >= 0; i--) {
        inv[i] = x * pre[i] % mod;
        x = x * a[i] % mod;
    }

    return inv;
}
```

### 高精度
``` c++
// 未处理负数的情况
struct bign {
    // 低位在前，高位在后
    // 例如 12345 存成 {5, 4, 3, 2, 1}
    vector<int> d;

    // 构造函数，默认是 0
    bign(long long num = 0) { *this = num; }
    bign(const string &s) { *this = s; }

    // 用 long long 赋值
    bign& operator = (long long num) {
        d.clear();
        if (num == 0) d.push_back(0);
        while (num > 0) {
            d.push_back(num % 10);
            num /= 10;
        }
        return *this;
    }

    // 用字符串赋值
    bign& operator = (const string &s) {
        d.clear();
        for (int i = (int)s.size() - 1; i >= 0; i--) {
            d.push_back(s[i] - '0');
        }
        trim();
        return *this;
    }

    // 去掉前导零
    void trim() {
        while (d.size() > 1 && d.back() == 0) d.pop_back();
    }

    // 输入
    friend istream& operator >> (istream &in, bign &x) {
        string s;
        in >> s;
        x = s;
        return in;
    }

    // 输出
    friend ostream& operator << (ostream &out, const bign &x) {
        for (int i = (int)x.d.size() - 1; i >= 0; i--) {
            out << x.d[i];
        }
        return out;
    }

    // 比较大小
    bool operator < (const bign &b) const {
        if (d.size() != b.d.size()) return d.size() < b.d.size();
        for (int i = (int)d.size() - 1; i >= 0; i--) {
            if (d[i] != b.d[i]) return d[i] < b.d[i];
        }
        return false; // 相等时返回 false
    }

    bool operator > (const bign &b) const { return b < *this; }
    bool operator <= (const bign &b) const { return !(b < *this); }
    bool operator >= (const bign &b) const { return !(*this < b); }
    bool operator == (const bign &b) const { return d == b.d; }
    bool operator != (const bign &b) const { return d != b.d; }

    // 高精度加法：A + B
    bign operator + (const bign &b) const {
        bign c;
        c.d.clear();

        int carry = 0;
        int n = d.size(), m = b.d.size();
        int len = max(n, m);

        for (int i = 0; i < len; i++) {
            int sum = carry;
            if (i < n) sum += d[i];
            if (i < m) sum += b.d[i];
            c.d.push_back(sum % 10);
            carry = sum / 10;
        }

        if (carry) c.d.push_back(carry);
        return c;
    }

    // 高精度减法：A - B，要求 A >= B
    bign operator - (const bign &b) const {
        bign c;
        c.d.clear();

        int borrow = 0;
        for (int i = 0; i < (int)d.size(); i++) {
            int cur = d[i] - borrow;
            if (i < (int)b.d.size()) cur -= b.d[i];

            if (cur < 0) {
                cur += 10;
                borrow = 1;
            } else {
                borrow = 0;
            }
            c.d.push_back(cur);
        }

        c.trim();
        return c;
    }

    // 高精度乘低精度：A * b
    bign operator * (int b) const {
        bign c;
        c.d.clear();

        if (b == 0) return bign(0);

        long long carry = 0;
        for (int i = 0; i < (int)d.size(); i++) {
            long long cur = 1LL * d[i] * b + carry;
            c.d.push_back(cur % 10);
            carry = cur / 10;
        }

        while (carry) {
            c.d.push_back(carry % 10);
            carry /= 10;
        }

        c.trim();
        return c;
    }

    // 高精度除低精度：A / b，返回商
    bign operator / (int b) const {
        bign c;
        c.d.resize(d.size());

        long long rem = 0;
        for (int i = (int)d.size() - 1; i >= 0; i--) {
            rem = rem * 10 + d[i];
            c.d[i] = rem / b;
            rem %= b;
        }

        c.trim();
        return c;
    }

    // 高精度除低精度：A % b，返回余数
    int operator % (int b) const {
        long long rem = 0;
        for (int i = (int)d.size() - 1; i >= 0; i--) {
            rem = rem * 10 + d[i];
            rem %= b;
        }
        return rem;
    }
};
```
